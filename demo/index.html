无alwaysShown参数，最大单位为0后不再显示; addZero:true会在分和秒前加零

notySeconds:60 在倒计时剩60秒时为倒计时容器div添加class="notifing",notifing为红色

$('#count-down1').fyCountDown({
  countdownSeconds: 65,
  addZero:true,
  notySeconds:60,
  notyClass:'notifing',
  onFinish: function(){
    //
  }
});

有alwaysShown参数，最大单位为0后依然显示。d为天，h为小时，m为分，d为秒； addZero:false不会在分和秒前加零

notySeconds:55 在倒计时剩55秒时为倒计时容器div添加class="notifing"

$('#count-down2').fyCountDown({
  countdownSeconds: 65,
  alwaysShown:'m',
  addZero:false,
  notySeconds:55,
  notyClass:'notifing',
  onFinish: function(){
    //
  }
});

倒计时结束回调

$('#count-down4').fyCountDown({
  countdownSeconds: 10,
  // alwaysShown:'d',
  addZero:true,
  notySeconds:5,
  notyClass:'notifing',
  onFinish: function(){
    console.log('我是第四个倒计时结束后执行的动作');
  }
});
