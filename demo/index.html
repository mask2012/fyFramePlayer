请注意，此demo必须在服务器环境下查看！

基本调用，循环播放，固定大小，每秒播放10朕

$('#player1').fyFramePlayer({
    rate: 10,
    fixedSize:true,
    movieName:'movieYaodou'
  });

播放完成后定格在最后一帧

$('#player4').fyFramePlayer({
    rate: 10,
    fixedSize:false,
    playTimes:1,
    stopOnFinish:true,
    movieName:'movieAnniu',
    onPlayEnd: function(){}
});

播放3次,不固定大小，根据屏幕宽度自适应大小,延时1.4秒播放,播放完毕后回调

$('#player2').fyFramePlayer({
    rate: 10,
    fixedSize:false,
    delay:1400,
    playTimes:3,
    movieName:'movieYaodou',
    onPlayEnd: function(){
        $('#player2_callback').html('播放完毕');
    }
  });

连续播放多个动画

$('#player3').fyFramePlayer({
    rate: 10,
    fixedSize:true,
    // playTimes:1,
    // stopOnFinish:true,
    movieName:'movieYaodou,movieAnniu,zhonghua',
    onPlayEnd: function(){}
  });
